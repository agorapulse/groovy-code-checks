18  package com.agorapulse.checks.gorm;

20  import org.codehaus.groovy.ast.*;

24  import org.codehaus.groovy.transform.AbstractASTTransformation;
25  import org.codehaus.groovy.transform.GroovyASTTransformation;
26  
27  import java.util.Arrays;
28  import java.util.HashSet;
29  import java.util.Optional;
30  import java.util.Set;
31  
32  @GroovyASTTransformation(phase = CompilePhase.FINALIZATION)
33  public class GormChecker extends AbstractASTTransformation {
34  
35      private static final Set<String> GORM_INSTANCE_METHOD_NAMES = new HashSet<>(Arrays.asList(
36              "instanceOf",
37              "lock",
38              "mutex",
39              "refresh",
40              "save",
41              "insert",
42              "merge",

48              "isDirty",
49              "getDirtyPropertyNames",
50              "getPersistentValue",
51              "getAssociationId",
52              "removeFrom",
53              "addTo"
54      ));
55  
56      private static final Set<String> GORM_STATIC_METHOD_NAMES = new HashSet<>(Arrays.asList(
57              "getGormPersistentEntity",
58              "getGormDynamicFinders",
59              "where",
60              "whereLazy",
61              "whereAny",
62              "findAll",

95  
96      @Override
97      public void visit(ASTNode[] nodes, SourceUnit source) {
98  
99          ClassCodeVisitorSupport support = new ClassCodeVisitorSupport() {
100 
101             private ClassNode currentClass;
102 
103             @Override
104             public void visitClass(ClassNode node) {
105                 if (node.getName() != null && node.getName().endsWith("DataService")) {
106                     // this might be a legacy bridge or an abstract implementation of the data service
107                     return;
108                 }
109                 this.currentClass = node;
110                 super.visitClass(node);
111                 this.currentClass = null;
112             }
113 
114             @Override
115             public void visitMethodCallExpression(MethodCallExpression call) {
116                 if (GORM_INSTANCE_METHOD_NAMES.contains(call.getMethodAsString()) || GORM_STATIC_METHOD_NAMES.contains(call.getMethodAsString())) {
117                     if (isEntityClass(getType(call))) {
118                         addError("Calling GORM methods is forbidden!", call);
119                     }
120                 }
121                 super.visitMethodCallExpression(call);
122             }
123 
124             @Override
125             public void visitStaticMethodCallExpression(StaticMethodCallExpression call) {
126                 if (GORM_STATIC_METHOD_NAMES.contains(call.getMethodAsString())) {
127                     if (isEntityClass(getType(call))) {
128                         addError("Calling GORM methods is forbidden!", call);
129                     }
130                 }
131                 super.visitStaticMethodCallExpression(call);
132             }
133 
134             @Override
135             protected SourceUnit getSourceUnit() {
136                 return source;
137             }
138 
139             private ClassNode getType(MethodCall call) {
140                 if (call instanceof ClassNode) {
141                     return (ClassNode) call;
142                 }
143 
144                 if (!(call instanceof Expression)) {
145                     return null;
146                 }
147                 Expression object = (Expression) call.getReceiver();
148 
149                 if (object.getColumnNumber() < 0 || object.getLineNumber() < 0) {
150                     // generated code
151                     return null;
152                 }
153 
154                 if (object instanceof VariableExpression) {
155                     Variable accessedVariable = ((VariableExpression) object).getAccessedVariable();
156                     if (accessedVariable != null) {
157                         return accessedVariable.getType();
158                     }
159                 }
160 
161                 if (object instanceof PropertyExpression) {
162                     PropertyExpression propertyExpression = (PropertyExpression) object;
163                     Expression objectExpression = propertyExpression.getObjectExpression();
164 
165                     if (objectExpression instanceof VariableExpression) {
166                         // instance field
167                         if (((VariableExpression) objectExpression).getName().equals("this") && currentClass != null) {
168                             PropertyNode property = currentClass.getProperty(propertyExpression.getPropertyAsString());
169                             return property.getField().getType();
170                         }
171                         return objectExpression.getType();
172                     }
173 
174                     if (objectExpression instanceof ClassExpression) {
175                         // static field
176                         ClassNode holder = objectExpression.getType();
177                         FieldNode field = holder.getField(propertyExpression.getPropertyAsString());
178                         return field.getType();
179                     }
180 
181                     if (objectExpression != null) {
182                         return objectExpression.getType();
183                     }
184                 }
185 
186                 if (object instanceof MethodCallExpression) {
187                     MethodCallExpression anotherCall = (MethodCallExpression) object;
188                     Expression objectExpression = anotherCall.getObjectExpression();
189                     if (objectExpression instanceof VariableExpression) {
190                         VariableExpression variable = (VariableExpression) objectExpression;
191                         if (variable.getName().equals("this") && currentClass != null) {
192                             // TODO: handle multiple matching methods
193                             Optional<MethodNode> method = currentClass.getMethods(anotherCall.getMethodAsString()).stream().findFirst();
194                             return method.map(MethodNode::getReturnType).orElse(null);
195                         }
196                         // TODO: handle multiple matching methods
197                         Optional<MethodNode> method = variable.getType().getMethods(anotherCall.getMethodAsString()).stream().findFirst();
198                         return method.map(MethodNode::getReturnType).orElse(null);
199                     } else {
200                         System.out.println(objectExpression);
201                     }
202                 }
203 
204                 if (object instanceof StaticMethodCallExpression) {
205                     StaticMethodCallExpression anotherCall = (StaticMethodCallExpression) object;
206                     Optional<MethodNode> method = anotherCall.getOwnerType().getMethods(anotherCall.getMethodAsString()).stream().findFirst();
207                     return method.map(MethodNode::getReturnType).orElse(null);
208                 }
209 
210                 return object.getType();
211             }
212 
213             private boolean isEntityClass(ClassNode type) {
214                 if (type == null) {
215                     return false;
216                 }
217 
218                 Set<ClassNode> allInterfaces = type.getAllInterfaces();
219 
220                 if (allInterfaces == null) {
221                     return false;
222                 }
223 
224                 return allInterfaces.stream().anyMatch(i -> {
225                     if (i != null && i.getName() != null) {
226                         return i.getName().equals("org.grails.datastore.gorm.GormEntityApi");
227                     }
228                     return false;
229                 });
230             }
231         };
232 
233         source.getAST().getClasses().forEach(support::visitClass);
234     }
